julia> # Шифр Цезаря

julia> function caesar_cipher(text::String, k::Int)
           encrypted_text = []
           for char in text
                   if 'a' <= char <= 'z'
                               push!(encrypted_text, Char((Int(char) - Int('a') + k) % 26 + Int('a')))
                                       elseif 'A' <= char <= 'Z'
                                                   push!(encrypted_text, Char((Int(char) - Int('A') + k) % 26 + Int('A')))
                                                           else
                   push!(encrypted_text, char)
                           end
           end
           return join(encrypted_text)
           end

julia> function atbash_cipher(text::String)
           encrypted_text = []
               for char in text
                       if 'a' <= char <= 'z'
                   push!(encrypted_text, Char(Int('z') - (Int(char) - Int('a'))))
                           elseif 'A' <= char <= 'Z'
                   push!(encrypted_text, Char(Int('Z') - (Int(char) - Int('A'))))
                           else
                   push!(encrypted_text, char)
                           end
           end
           return join(encrypted_text)
           end
